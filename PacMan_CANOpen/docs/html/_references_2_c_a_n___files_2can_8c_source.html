86 /* Formatted frame structure has identical layout as TX/RX buffer registers.
87  This allows for direct copy to/from TX/RX buffer. The two reserved bits in TX
88  buffer are used in the frame structure to store the self_reception and
89  single_shot flags. */
90 typedef union {
91  struct {
92  struct {
93  uint8_t dlc: 4; //Data length code (0 to 8) of the frame
94  uint8_t self_reception: 1; //This frame should be transmitted using self reception command
95  uint8_t single_shot: 1; //This frame should be transmitted using single shot command
96  uint8_t rtr: 1; //This frame is a remote transmission request
97  uint8_t frame_format: 1; //Format of the frame (1 = extended, 0 = standard)
98  };
99  union {
100  struct {
101  uint8_t id[FRAME_STD_ID_LEN]; //11 bit standard frame identifier
102  uint8_t data[FRAME_MAX_DATA_LEN]; //Data bytes (0 to 8)
103  uint8_t reserved8[2];
104  } standard;
105  struct {
106  uint8_t id[FRAME_EXTD_ID_LEN]; //29 bit extended frame identifier
107  uint8_t data[FRAME_MAX_DATA_LEN]; //Data bytes (0 to 8)
108  } extended;
109  };
110  };
111  uint8_t bytes[FRAME_MAX_LEN];
112 } can_frame_t;

145 static inline esp_err_t can_enter_reset_mode()
146 {
147  /* Enter reset mode (required to write to configuration registers). Reset mode
148  also prevents all CAN activity on the current module and is automatically
149  set upon entering a BUS-OFF condition. */
150  CAN.mode_reg.reset = 1; //Set reset mode bit
151  CAN_CHECK(CAN.mode_reg.reset == 1, ESP_ERR_INVALID_STATE); //Check bit was set
152  return ESP_OK;
153 }
154 
155 static inline esp_err_t can_exit_reset_mode()
156 {
157  /* Exiting reset mode will return the CAN module to operating mode. Reset mode
158  must also be exited in order to trigger BUS-OFF recovery sequence. */
159  CAN.mode_reg.reset = 0; //Exit reset mode
160  CAN_CHECK(CAN.mode_reg.reset == 0, ESP_ERR_INVALID_STATE); //Check bit was reset
161  return ESP_OK;
162 }

170 static inline void can_config_mode(can_mode_t mode)
171 {
172  //Configure CAN mode of operation
173  can_mode_reg_t mode_reg;
174  mode_reg.val = CAN.mode_reg.val; //Get current value of mode register
175  if (mode == CAN_MODE_NO_ACK) {
176  mode_reg.self_test = 1;
177  mode_reg.listen_only = 0;
178  } else if (mode == CAN_MODE_LISTEN_ONLY) {
179  mode_reg.self_test = 0;
180  mode_reg.listen_only = 1;
181  } else {
182  //Default to normal operating mode
183  mode_reg.self_test = 0;
184  mode_reg.listen_only = 0;
185  }
186  CAN.mode_reg.val = mode_reg.val; //Write back modified value to register
187 }

195 static inline void can_config_bus_timing(uint32_t brp, uint32_t sjw, uint32_t tseg_1, uint32_t tseg_2, bool triple_sampling)
196 {
197  /* Configure bus/bit timing of CAN peripheral.
198  - BRP (even from 2 to 128) divide APB to CAN system clock (T_scl)
199  - SJW (1 to 4) is number of T_scl to shorten/lengthen for bit synchronization
200  - TSEG_1 (1 to 16) is number of T_scl in a bit time before sample point
201  - TSEG_2 (1 to 8) is number of T_scl in a bit time after sample point
202  - triple_sampling will cause each bit time to be sampled 3 times*/
203  can_bus_tim_0_reg_t timing_reg_0;
204  can_bus_tim_1_reg_t timing_reg_1;
205  timing_reg_0.baud_rate_prescaler = (brp / 2) - 1;
206  timing_reg_0.sync_jump_width = sjw - 1;
207  timing_reg_1.time_seg_1 = tseg_1 - 1;
208  timing_reg_1.time_seg_2 = tseg_2 - 1;
209  timing_reg_1.sampling = triple_sampling;
210  CAN.bus_timing_0_reg.val = timing_reg_0.val;
211  CAN.bus_timing_1_reg.val = timing_reg_1.val;
212 }

214 static inline void can_config_error(int err_warn_lim, int rx_err_cnt, int tx_err_cnt)
215 {
216  /* Set error warning limit, RX error counter, and TX error counter. Note that
217  forcibly setting RX/TX error counters will incur the expected status changes
218  and interrupts as soon as reset mode exits. */
219  if (err_warn_lim >= 0 && err_warn_lim <= UINT8_MAX) {
220  //Defaults to 96 after hardware reset.
221  CAN.error_warning_limit_reg.byte = err_warn_lim;
222  }
223  if (rx_err_cnt >= 0 && rx_err_cnt <= UINT8_MAX) {
224  //Defaults to 0 after hardware reset.
225  CAN.rx_error_counter_reg.byte = rx_err_cnt;
226  }
227  if (tx_err_cnt >= 0 && tx_err_cnt <= UINT8_MAX) {
228  //Defaults to 0 after hardware reset, and 127 after BUS-OFF event
229  CAN.tx_error_counter_reg.byte = tx_err_cnt;
230  }
231 }

233 static inline void can_config_acceptance_filter(uint32_t code, uint32_t mask, bool single_filter)
234 {
235  //Set filter mode
236  CAN.mode_reg.acceptance_filter = (single_filter) ? 1 : 0;
237  //Swap code and mask to match big endian registers
238  uint32_t code_swapped = __builtin_bswap32(code);
239  uint32_t mask_swapped = __builtin_bswap32(mask);
240  for (int i = 0; i < 4; i++) {
241  CAN.acceptance_filter.code_reg[i].byte = ((code_swapped >> (i * 8)) & 0xFF);
242  CAN.acceptance_filter.mask_reg[i].byte = ((mask_swapped >> (i * 8)) & 0xFF);
243  }
244 }

246 static inline void can_config_clk_out(uint32_t divider)
247 {
248  /* Configure CLKOUT. CLKOUT is a pre-scaled version of APB CLK. Divider can be
249  1, or any even number from 2 to 14. Set to out of range value (0) to disable
250  CLKOUT. */
251  can_clk_div_reg_t clock_divider_reg;
252  clock_divider_reg.val = CAN.clock_divider_reg.val;
253  if (divider >= 2 && divider <= 14) {
254  clock_divider_reg.clock_off = 0;
255  clock_divider_reg.clock_divider = (divider / 2) - 1;
256  } else if (divider == 1) {
257  clock_divider_reg.clock_off = 0;
258  clock_divider_reg.clock_divider = 7;
259  } else {
260  clock_divider_reg.clock_off = 1;
261  clock_divider_reg.clock_divider = 0;
262  }
263  CAN.clock_divider_reg.val = clock_divider_reg.val;
264 }

286  }

339 static void can_alert_handler(uint32_t alert_code, int *alert_req)
340 {
341  if (p_can_obj->alerts_enabled & alert_code) {
342  //Signify alert has occurred
343  CAN_SET_FLAG(p_can_obj->alerts_triggered, alert_code);
344  *alert_req = 1;
345  if (p_can_obj->alerts_enabled & CAN_ALERT_AND_LOG) {
346  if (alert_code >= ALERT_LOG_LEVEL_ERROR) {
347  ESP_EARLY_LOGE(CAN_TAG, "Alert %d", alert_code);
348  } else if (alert_code >= ALERT_LOG_LEVEL_WARNING) {
349  ESP_EARLY_LOGW(CAN_TAG, "Alert %d", alert_code);
350  } else {
351  ESP_EARLY_LOGI(CAN_TAG, "Alert %d", alert_code);
352  }
353  }
354  }
355 }

357 static void can_intr_handler_err_warn(can_status_reg_t *status, BaseType_t *task_woken, int *alert_req)
358 {
359  if (status->bus) {
360  if (status->error) {
361  //Bus-Off condition. TEC should set and held at 127, REC should be 0, reset mode entered
362  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_BUS_OFF);
363  /* Note: REC is still allowed to increase during bus-off. REC > err_warn
364  can prevent "bus recovery complete" interrupt from occurring. Set to
365  listen only mode to freeze REC. */
366  can_config_mode(CAN_MODE_LISTEN_ONLY);
367  can_alert_handler(CAN_ALERT_BUS_OFF, alert_req);
368  } else {
369  //Bus-recovery in progress. TEC has dropped below error warning limit
370  can_alert_handler(CAN_ALERT_RECOVERY_IN_PROGRESS, alert_req);
371  }
372  } else {
373  if (status->error) {
374  //TEC or REC surpassed error warning limit
375  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_ERR_WARN);
376  can_alert_handler(CAN_ALERT_ABOVE_ERR_WARN, alert_req);
377  } else if (p_can_obj->control_flags & CTRL_FLAG_RECOVERING) {
378  //Bus recovery complete.
379  can_enter_reset_mode();
380  //Reset and set flags to the equivalent of the stopped state
381  CAN_RESET_FLAG(p_can_obj->control_flags, CTRL_FLAG_RECOVERING | CTRL_FLAG_ERR_WARN |
382  CTRL_FLAG_ERR_PASSIVE | CTRL_FLAG_BUS_OFF |
383  CTRL_FLAG_TX_BUFF_OCCUPIED);
384  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_STOPPED);
385  can_alert_handler(CAN_ALERT_BUS_RECOVERED, alert_req);
386  } else {
387  //TEC and REC are both below error warning
388  CAN_RESET_FLAG(p_can_obj->control_flags, CTRL_FLAG_ERR_WARN);
389  can_alert_handler(CAN_ALERT_BELOW_ERR_WARN, alert_req);
390  }
391  }
392 }

394 static void can_intr_handler_err_passive(int *alert_req)
395 {
396  uint32_t tec, rec;
397  can_get_error_counters(&tec, &rec);
398  if (tec >= DRIVER_DEFAULT_ERR_PASS_CNT || rec >= DRIVER_DEFAULT_ERR_PASS_CNT) {
399  //Entered error passive
400  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_ERR_PASSIVE);
401  can_alert_handler(CAN_ALERT_ERR_PASS, alert_req);
402  } else {
403  //Returned to error active
404  CAN_RESET_FLAG(p_can_obj->control_flags, CTRL_FLAG_ERR_PASSIVE);
405  can_alert_handler(CAN_ALERT_ERR_ACTIVE, alert_req);
406  }
407 }

445 static void can_intr_handler_tx(can_status_reg_t *status, int *alert_req)
446 {
447  //Handle previously transmitted frame
448  if (status->tx_complete) {
449  can_alert_handler(CAN_ALERT_TX_SUCCESS, alert_req);
450  } else {
451  p_can_obj->tx_failed_count++;
452  can_alert_handler(CAN_ALERT_TX_FAILED, alert_req);
453  }
454 
455  //Update TX message count
456  p_can_obj->tx_msg_count--;
457  configASSERT(p_can_obj->tx_msg_count >= 0); //Sanity check
458 
459  //Check if there are more frames to transmit
460  if (p_can_obj->tx_msg_count > 0 && p_can_obj->tx_queue != NULL) {
461  can_frame_t frame;
462  configASSERT(xQueueReceiveFromISR(p_can_obj->tx_queue, &frame, NULL) == pdTRUE);
463  can_set_tx_buffer_and_transmit(&frame);
464  } else {
465  //No more frames to transmit
466  CAN_RESET_FLAG(p_can_obj->control_flags, CTRL_FLAG_TX_BUFF_OCCUPIED);
467  can_alert_handler(CAN_ALERT_TX_IDLE, alert_req);
468  }
469 }

471 static void can_intr_handler_main(void *arg)
472 {
473  BaseType_t task_woken = pdFALSE;
474  int alert_req = 0;
475  can_status_reg_t status;
476  can_intr_reg_t intr_reason;
477 
478  CAN_ENTER_CRITICAL();
479  status.val = can_get_status();
480  intr_reason.val = (p_can_obj != NULL) ? can_get_interrupt_reason() : 0; //Incase intr occurs whilst driver is being uninstalled
481 
482  //Handle error counter related interrupts
483  if (intr_reason.err_warn) {
484  //Triggers when Bus-Status or Error-status bits change
485  can_intr_handler_err_warn(&status, &task_woken, &alert_req);
486  }
487  if (intr_reason.err_passive) {
488  //Triggers when entering/returning error passive/active state
489  can_intr_handler_err_passive(&alert_req);
490  }
491 
492  //Handle other error interrupts
493  if (intr_reason.bus_err) {
494  //Triggers when an error (Bit, Stuff, CRC, Form, ACK) occurs on the CAN bus
495  can_intr_handler_bus_err(&alert_req);
496  }
497  if (intr_reason.arb_lost) {
498  //Triggers when arbitration is lost
499  can_intr_handler_arb_lost(&alert_req);
500  }
501 
502  //Handle TX/RX interrupts
503  if (intr_reason.rx) {
504  //Triggers when RX buffer has one or more frames. Disabled if RX Queue length = 0
505  can_intr_handler_rx(&task_woken, &alert_req);
506  }
507  if (intr_reason.tx) {
508  //Triggers when TX buffer becomes free after a transmission
509  can_intr_handler_tx(&status, &alert_req);
510  }
511  /* Todo: Check possible bug where transmitting self reception request then
512  clearing rx buffer will cancel the transmission. */
513  CAN_EXIT_CRITICAL();
514 
515  if (p_can_obj->alert_semphr != NULL && alert_req) {
516  //Give semaphore if alerts were triggered
517  xSemaphoreGiveFromISR(p_can_obj->alert_semphr, &task_woken);
518  }
519  if (task_woken == pdTRUE) {
520  portYIELD_FROM_ISR();
521  }
522 }

526 static void can_format_frame(uint32_t id, uint8_t dlc, const uint8_t *data, uint32_t flags, can_frame_t *tx_frame)
527 {
528  /* This function encodes a message into a frame structure. The frame structure has
529  an identical layout to the TX buffer, allowing the frame structure to be directly
530  copied into TX buffer. */
531  //Set frame information
532  tx_frame->dlc = dlc;
533  tx_frame->rtr = (flags & CAN_MSG_FLAG_RTR) ? 1 : 0;
534  tx_frame->frame_format = (flags & CAN_MSG_FLAG_EXTD) ? 1 : 0;
535  tx_frame->self_reception = (flags & CAN_MSG_FLAG_SELF) ? 1 : 0;
536  tx_frame->single_shot = (flags & CAN_MSG_FLAG_SS) ? 1 : 0;
537 
538  //Set ID
539  int id_len = (flags & CAN_MSG_FLAG_EXTD) ? FRAME_EXTD_ID_LEN : FRAME_STD_ID_LEN;
540  uint8_t *id_buffer = (flags & CAN_MSG_FLAG_EXTD) ? tx_frame->extended.id : tx_frame->standard.id;
541  //Split ID into 4 or 2 bytes, and turn into big-endian with left alignment (<< 3 or 5)
542  uint32_t id_temp = (flags & CAN_MSG_FLAG_EXTD) ? __builtin_bswap32((id & CAN_EXTD_ID_MASK) << 3) : //((id << 3) >> 8*(3-i))
543  __builtin_bswap16((id & CAN_STD_ID_MASK) << 5); //((id << 5) >> 8*(1-i))
544  for (int i = 0; i < id_len; i++) {
545  id_buffer[i] = (id_temp >> (8 * i)) & 0xFF; //Copy big-endian ID byte by byte
546  }
547 
548  //Set Data.
549  uint8_t *data_buffer = (flags & CAN_MSG_FLAG_EXTD) ? tx_frame->extended.data : tx_frame->standard.data;
550  for (int i = 0; (i < dlc) && (i < FRAME_MAX_DATA_LEN); i++) { //Handle case where dlc is > 8
551  data_buffer[i] = data[i];
552  }
553 }

555 static void can_parse_frame(can_frame_t *rx_frame, uint32_t *id, uint8_t *dlc, uint8_t *data, uint32_t *flags)
556 {
557  //This function decodes a frame structure into it's constituent components.
558 
559  //Copy frame information
560  *dlc = rx_frame->dlc;
561  *flags = 0;
562  *flags |= (rx_frame->dlc > FRAME_MAX_DATA_LEN) ? CAN_MSG_FLAG_DLC_NON_COMP : 0;
563  *flags |= (rx_frame->rtr) ? CAN_MSG_FLAG_RTR : 0;
564  *flags |= (rx_frame->frame_format) ? CAN_MSG_FLAG_EXTD : 0;
565 
566  //Copy ID
567  int id_len = (rx_frame->frame_format) ? FRAME_EXTD_ID_LEN : FRAME_STD_ID_LEN;
568  uint8_t *id_buffer = (rx_frame->frame_format) ? rx_frame->extended.id : rx_frame->standard.id;
569  uint32_t id_temp = 0;
570  for (int i = 0; i < id_len; i++) {
571  id_temp |= id_buffer[i] << (8 * i); //Copy big-endian ID byte by byte
572  }
573  //Revert endianness of 4 or 2 byte ID, and shift into 29 or 11 bit ID
574  id_temp = (rx_frame->frame_format) ? (__builtin_bswap32(id_temp) >> 3) : //((byte[i] << 8*(3-i)) >> 3)
575  (__builtin_bswap16(id_temp) >> 5); //((byte[i] << 8*(1-i)) >> 5)
576  *id = id_temp & ((rx_frame->frame_format) ? CAN_EXTD_ID_MASK : CAN_STD_ID_MASK);
577 
578  //Copy data
579  uint8_t *data_buffer = (rx_frame->frame_format) ? rx_frame->extended.data : rx_frame->standard.data;
580  for (int i = 0; (i < rx_frame->dlc) && (i < FRAME_MAX_DATA_LEN); i++) {
581  data[i] = data_buffer[i];
582  }
583  //Set remaining bytes of data to 0
584  for (int i = rx_frame->dlc; i < FRAME_MAX_DATA_LEN; i++) {
585  data[i] = 0;
586  }
587 }

815 esp_err_t can_transmit(const can_message_t *message, TickType_t ticks_to_wait)
816 {
817  //Check arguments
818  CAN_CHECK(p_can_obj != NULL, ESP_ERR_INVALID_STATE);
819  CAN_CHECK(message != NULL, ESP_ERR_INVALID_ARG);
820  CAN_CHECK((message->data_length_code <= FRAME_MAX_DATA_LEN) || (message->flags & CAN_MSG_FLAG_DLC_NON_COMP), ESP_ERR_INVALID_ARG);
821 
822  CAN_ENTER_CRITICAL();
823  //Check State
824  CAN_CHECK_FROM_CRIT(!(p_can_obj->control_flags & CTRL_FLAG_LISTEN_ONLY), ESP_ERR_NOT_SUPPORTED);
825  CAN_CHECK_FROM_CRIT(!(p_can_obj->control_flags & (CTRL_FLAG_STOPPED | CTRL_FLAG_BUS_OFF)), ESP_ERR_INVALID_STATE);
826  //Format frame
827  esp_err_t ret = ESP_FAIL;
828  can_frame_t tx_frame;
829  can_format_frame(message->identifier, message->data_length_code, message->data, message->flags, &tx_frame);
830  //Check if frame can be sent immediately
831  if ((p_can_obj->tx_msg_count == 0) && !(p_can_obj->control_flags & CTRL_FLAG_TX_BUFF_OCCUPIED)) {
832  //No other frames waiting to transmit. Bypass queue and transmit immediately
833  can_set_tx_buffer_and_transmit(&tx_frame);
834  p_can_obj->tx_msg_count++;
835  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_TX_BUFF_OCCUPIED);
836  ret = ESP_OK;
837  }
838  CAN_EXIT_CRITICAL();
839 
840  if (ret != ESP_OK) {
841  if (p_can_obj->tx_queue == NULL) {
842  //TX Queue is disabled and TX buffer is occupied, message was not sent
843  ret = ESP_FAIL;
844  } else if (xQueueSend(p_can_obj->tx_queue, &tx_frame, ticks_to_wait) == pdTRUE) {
845  //Copied to TX Queue
846  CAN_ENTER_CRITICAL();
847  if (p_can_obj->control_flags & (CTRL_FLAG_STOPPED | CTRL_FLAG_BUS_OFF)) {
848  //TX queue was reset (due to stop/bus_off), remove copied frame from queue to prevent transmission
849  configASSERT(xQueueReceive(p_can_obj->tx_queue, &tx_frame, 0) == pdTRUE);
850  ret = ESP_ERR_INVALID_STATE;
851  } else if ((p_can_obj->tx_msg_count == 0) && !(p_can_obj->control_flags & CTRL_FLAG_TX_BUFF_OCCUPIED)) {
852  //TX buffer was freed during copy, manually trigger transmission
853  configASSERT(xQueueReceive(p_can_obj->tx_queue, &tx_frame, 0) == pdTRUE);
854  can_set_tx_buffer_and_transmit(&tx_frame);
855  p_can_obj->tx_msg_count++;
856  CAN_SET_FLAG(p_can_obj->control_flags, CTRL_FLAG_TX_BUFF_OCCUPIED);
857  ret = ESP_OK;
858  } else {
859  //Frame was copied to queue, waiting to be transmitted
860  p_can_obj->tx_msg_count++;
861  ret = ESP_OK;
862  }
863  CAN_EXIT_CRITICAL();
864  } else {
865  //Timed out waiting for free space on TX queue
866  ret = ESP_ERR_TIMEOUT;
867  }
868  }
869  return ret;
870 }
