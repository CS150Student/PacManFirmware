33 /*
34  * Read received message from CAN module.
35  *
36  * Function will be called (by CAN receive interrupt) every time, when CAN
37  * message with correct identifier will be received. For more information and
38  * description of parameters see file CO_driver.h.
39  */
40 static void CO_SYNC_receive(void *object, const CO_CANrxMsg_t *msg){
41  CO_SYNC_t *SYNC;
42  uint8_t operState;
43 
44  SYNC = (CO_SYNC_t*)object; /* this is the correct pointer type of the first argument */
45  operState = *SYNC->operatingState;
46 
47  if((operState == CO_NMT_OPERATIONAL) || (operState == CO_NMT_PRE_OPERATIONAL)){
48  if(SYNC->counterOverflowValue == 0){
49  if(msg->DLC == 0U){
50  SET_CANrxNew(SYNC->CANrxNew);
51  }
52  else{
53  SYNC->receiveError = (uint16_t)msg->DLC | 0x0100U;
54  }
55  }
56  else{
57  if(msg->DLC == 1U){
58  SYNC->counter = msg->data[0];
59  SET_CANrxNew(SYNC->CANrxNew);
60  }
61  else{
62  SYNC->receiveError = (uint16_t)msg->DLC | 0x0200U;
63  }
64  }
65  if(IS_CANrxNew(SYNC->CANrxNew)) {
66  SYNC->CANrxToggle = SYNC->CANrxToggle ? false : true;
67  }
68  }
69 }

72 /*
73  * Function for accessing _COB ID SYNC Message_ (index 0x1005) from SDO server.
74  *
75  * For more information see file CO_SDO.h.
76  */
77 static CO_SDO_abortCode_t CO_ODF_1005(CO_ODF_arg_t *ODF_arg){
78  CO_SYNC_t *SYNC;
79  uint32_t value;
80  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
81 
82  SYNC = (CO_SYNC_t*) ODF_arg->object;
83  value = CO_getUint32(ODF_arg->data);
84 
85  if(!ODF_arg->reading){
86  uint8_t configureSyncProducer = 0;
87 
88  /* only 11-bit CAN identifier is supported */
89  if(value & 0x20000000UL){
90  ret = CO_SDO_AB_INVALID_VALUE;
91  }
92  else{
93  /* is 'generate Sync messge' bit set? */
94  if(value & 0x40000000UL){
95  /* if bit was set before, value can not be changed */
96  if(SYNC->isProducer){
97  ret = CO_SDO_AB_DATA_DEV_STATE;
98  }
99  else{
100  configureSyncProducer = 1;
101  }
102  }
103  }
104 
105  /* configure sync producer and consumer */
106  if(ret == CO_SDO_AB_NONE){
107  SYNC->COB_ID = (uint16_t)(value & 0x7FFU);
108 
109  if(configureSyncProducer){
110  uint8_t len = 0U;
111  if(SYNC->counterOverflowValue != 0U){
112  len = 1U;
113  SYNC->counter = 0U;
114  SYNC->timer = 0U;
115  }
116  SYNC->CANtxBuff = CO_CANtxBufferInit(
117  SYNC->CANdevTx, /* CAN device */
118  SYNC->CANdevTxIdx, /* index of specific buffer inside CAN module */
119  SYNC->COB_ID, /* CAN identifier */
120  0, /* rtr */
121  len, /* number of data bytes */
122  0); /* synchronous message flag bit */
123  SYNC->isProducer = true;
124  }
125  else{
126  SYNC->isProducer = false;
127  }
128  printf("Sync abourt Print: %u ", SYNC->COB_ID);
129 
130  CO_CANrxBufferInit(
131  SYNC->CANdevRx, /* CAN device */
132  SYNC->CANdevRxIdx, /* rx buffer index */
133  SYNC->COB_ID, /* CAN identifier */
134  0x7FF, /* mask */
135  0, /* rtr */
136  (void*)SYNC, /* object passed to receive function */
137  CO_SYNC_receive); /* this function will process received message */
138  }
139  }
140 
141  return ret;
142 }

145 /*
146  * Function for accessing _Communication cycle period_ (index 0x1006) from SDO server.
147  *
148  * For more information see file CO_SDO.h.
149  */
150 static CO_SDO_abortCode_t CO_ODF_1006(CO_ODF_arg_t *ODF_arg){
151  CO_SYNC_t *SYNC;
152  uint32_t value;
153  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
154 
155  SYNC = (CO_SYNC_t*) ODF_arg->object;
156  value = CO_getUint32(ODF_arg->data);
157 
158  if(!ODF_arg->reading){
159  /* period transition from 0 to something */
160  if((SYNC->periodTime == 0) && (value != 0)){
161  SYNC->counter = 0;
162  }
163 
164  SYNC->periodTime = value;
165  SYNC->periodTimeoutTime = (value / 2U) * 3U;
166  /* overflow? */
167  if(SYNC->periodTimeoutTime < value){
168  SYNC->periodTimeoutTime = 0xFFFFFFFFUL;
169  }
170 
171  SYNC->timer = 0;
172  }
173 
174  return ret;
175 }

221 CO_ReturnError_t CO_SYNC_init(
222  CO_SYNC_t *SYNC,
223  CO_EM_t *em,
224  CO_SDO_t *SDO,
225  uint8_t *operatingState,
226  uint32_t COB_ID_SYNCMessage,
227  uint32_t communicationCyclePeriod,
228  uint8_t synchronousCounterOverflowValue,
229  CO_CANmodule_t *CANdevRx,
230  uint16_t CANdevRxIdx,
231  CO_CANmodule_t *CANdevTx,
232  uint16_t CANdevTxIdx)
233 {
234  uint8_t len = 0;
235 
236  /* verify arguments */
237  if(SYNC==NULL || em==NULL || SDO==NULL || operatingState==NULL ||
238  CANdevRx==NULL || CANdevTx==NULL){
239  return CO_ERROR_ILLEGAL_ARGUMENT;
240  }
241 
242  /* Configure object variables */
243  SYNC->isProducer = (COB_ID_SYNCMessage&0x40000000L) ? true : false;
244  SYNC->COB_ID = COB_ID_SYNCMessage&0x7FF;
245 
246  SYNC->periodTime = communicationCyclePeriod;
247  SYNC->periodTimeoutTime = communicationCyclePeriod / 2 * 3;
248  /* overflow? */
249  if(SYNC->periodTimeoutTime < communicationCyclePeriod) SYNC->periodTimeoutTime = 0xFFFFFFFFL;
250 
251  SYNC->counterOverflowValue = synchronousCounterOverflowValue;
252  if(synchronousCounterOverflowValue) len = 1;
253 
254  SYNC->curentSyncTimeIsInsideWindow = true;
255 
256  CLEAR_CANrxNew(SYNC->CANrxNew);
257  SYNC->CANrxToggle = false;
258  SYNC->timer = 0;
259  SYNC->counter = 0;
260  SYNC->receiveError = 0U;
261 
262  SYNC->em = em;
263  SYNC->operatingState = operatingState;
264 
265  SYNC->CANdevRx = CANdevRx;
266  SYNC->CANdevRxIdx = CANdevRxIdx;
267 
268  /* Configure Object dictionary entry at index 0x1005, 0x1006 and 0x1019 */
269  CO_OD_configure(SDO, OD_H1005_COBID_SYNC, CO_ODF_1005, (void*)SYNC, 0, 0);
270  CO_OD_configure(SDO, OD_H1006_COMM_CYCL_PERIOD, CO_ODF_1006, (void*)SYNC, 0, 0);
271  CO_OD_configure(SDO, OD_H1019_SYNC_CNT_OVERFLOW, CO_ODF_1019, (void*)SYNC, 0, 0);
272  printf("\nsync init print: %u ", CANdevRxIdx);
273  /* configure SYNC CAN reception */
274  CO_CANrxBufferInit(
275  CANdevRx, /* CAN device */
276  CANdevRxIdx, /* rx buffer index */
277  SYNC->COB_ID, /* CAN identifier */
278  0x7FF, /* mask */
279  0, /* rtr */
280  (void*)SYNC, /* object passed to receive function */
281  CO_SYNC_receive); /* this function will process received message */
282 
283  /* configure SYNC CAN transmission */
284  SYNC->CANdevTx = CANdevTx;
285  SYNC->CANdevTxIdx = CANdevTxIdx;
286  SYNC->CANtxBuff = CO_CANtxBufferInit(
287  CANdevTx, /* CAN device */
288  CANdevTxIdx, /* index of specific buffer inside CAN module */
289  SYNC->COB_ID, /* CAN identifier */
290  0, /* rtr */
291  len, /* number of data bytes */
292  0); /* synchronous message flag bit */
293 
294  return CO_ERROR_NO;
295 }

299 uint8_t CO_SYNC_process(
300  CO_SYNC_t *SYNC,
301  uint32_t timeDifference_us,
302  uint32_t ObjDict_synchronousWindowLength)
303 {
304  uint8_t ret = 0;
305  uint32_t timerNew;
306 
307  if(*SYNC->operatingState == CO_NMT_OPERATIONAL || *SYNC->operatingState == CO_NMT_PRE_OPERATIONAL){
308  /* update sync timer, no overflow */
309  timerNew = SYNC->timer + timeDifference_us;
310  if(timerNew > SYNC->timer) SYNC->timer = timerNew;
311 
312  /* was SYNC just received */
313  if(IS_CANrxNew(SYNC->CANrxNew)){
314  SYNC->timer = 0;
315  ret = 1;
316  CLEAR_CANrxNew(SYNC->CANrxNew);
317  }
318 
319  /* SYNC producer */
320  if(SYNC->isProducer && SYNC->periodTime){
321  if(SYNC->timer >= SYNC->periodTime){
322  if(++SYNC->counter > SYNC->counterOverflowValue) SYNC->counter = 1;
323  SYNC->timer = 0;
324  ret = 1;
325  SYNC->CANrxToggle = SYNC->CANrxToggle ? false : true;
326  SYNC->CANtxBuff->data[0] = SYNC->counter;
327  CO_CANsend(SYNC->CANdevTx, SYNC->CANtxBuff);
328  }
329  }
330 
331  /* Synchronous PDOs are allowed only inside time window */
332  if(ObjDict_synchronousWindowLength){
333  if(SYNC->timer > ObjDict_synchronousWindowLength){
334  if(SYNC->curentSyncTimeIsInsideWindow){
335  ret = 2;
336  }
337  SYNC->curentSyncTimeIsInsideWindow = false;
338  }
339  else{
340  SYNC->curentSyncTimeIsInsideWindow = true;
341  }
342  }
343  else{
344  SYNC->curentSyncTimeIsInsideWindow = true;
345  }
346 
347  /* Verify timeout of SYNC */
348  if(SYNC->periodTime && SYNC->timer > SYNC->periodTimeoutTime && *SYNC->operatingState == CO_NMT_OPERATIONAL)
349  CO_errorReport(SYNC->em, CO_EM_SYNC_TIME_OUT, CO_EMC_COMMUNICATION, SYNC->timer);
350  }
351  else {
352  CLEAR_CANrxNew(SYNC->CANrxNew);
353  }
354 
355  /* verify error from receive function */
356  if(SYNC->receiveError != 0U){
357  CO_errorReport(SYNC->em, CO_EM_SYNC_LENGTH, CO_EMC_SYNC_DATA_LENGTH, (uint32_t)SYNC->receiveError);
358  SYNC->receiveError = 0U;
359  }
360 
361  return ret;
362 }
