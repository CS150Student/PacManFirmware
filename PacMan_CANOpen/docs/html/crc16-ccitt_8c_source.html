85 unsigned short crc16_ccitt(
86  const unsigned char block[],
87  unsigned int blockLength,
88  unsigned short crc)
89 {
90  unsigned int i;
91 
92  for(i=0U; i<blockLength; i++){
93  unsigned short tmp = (crc >> 8) ^ (unsigned short) block[i];
94  crc = ((unsigned short)(crc << 8U)) ^ crc16_ccitt_table[tmp];
95  }
96  return crc;
97 }
