146 static esp_err_t gpio_output_disable(gpio_num_t gpio_num)
147 {
148  GPIO_CHECK(GPIO_IS_VALID_GPIO(gpio_num), "GPIO number error", ESP_ERR_INVALID_ARG);
149  if (gpio_num < 32) {
150  GPIO.enable_w1tc = (0x1 << gpio_num);
151  } else {
152  GPIO.enable1_w1tc.data = (0x1 << (gpio_num - 32));
153  }
154 
155  // Ensure no other output signal is routed via GPIO matrix to this pin
156  REG_WRITE(GPIO_FUNC0_OUT_SEL_CFG_REG + (gpio_num * 4),
157  SIG_GPIO_OUT_IDX);
158 
159  return ESP_OK;
160 }

202 esp_err_t gpio_set_pull_mode(gpio_num_t gpio_num, gpio_pull_mode_t pull)
203 {
204  GPIO_CHECK(GPIO_IS_VALID_GPIO(gpio_num), "GPIO number error", ESP_ERR_INVALID_ARG);
205  GPIO_CHECK(pull <= GPIO_FLOATING, "GPIO pull mode error", ESP_ERR_INVALID_ARG);
206  esp_err_t ret = ESP_OK;
207  switch (pull) {
208  case GPIO_PULLUP_ONLY:
209  gpio_pulldown_dis(gpio_num);
210  gpio_pullup_en(gpio_num);
211  break;
212  case GPIO_PULLDOWN_ONLY:
213  gpio_pulldown_en(gpio_num);
214  gpio_pullup_dis(gpio_num);
215  break;
216  case GPIO_PULLUP_PULLDOWN:
217  gpio_pulldown_en(gpio_num);
218  gpio_pullup_en(gpio_num);
219  break;
220  case GPIO_FLOATING:
221  gpio_pulldown_dis(gpio_num);
222  gpio_pullup_dis(gpio_num);
223  break;
224  default:
225  ESP_LOGE(GPIO_TAG, "Unknown pull up/down mode,gpio_num=%u,pull=%u", gpio_num, pull);
226  ret = ESP_ERR_INVALID_ARG;
227  break;
228  }
229  return ret;
230 }

232 esp_err_t gpio_set_direction(gpio_num_t gpio_num, gpio_mode_t mode)
233 {
234  GPIO_CHECK(GPIO_IS_VALID_GPIO(gpio_num), "GPIO number error", ESP_ERR_INVALID_ARG);
235  if (gpio_num >= 34 && (mode & GPIO_MODE_DEF_OUTPUT)) {
236  ESP_LOGE(GPIO_TAG, "io_num=%d can only be input", gpio_num);
237  return ESP_ERR_INVALID_ARG;
238  }
239  esp_err_t ret = ESP_OK;
240  if (mode & GPIO_MODE_DEF_INPUT) {
241  PIN_INPUT_ENABLE(GPIO_PIN_MUX_REG[gpio_num]);
242  } else {
243  PIN_INPUT_DISABLE(GPIO_PIN_MUX_REG[gpio_num]);
244  }
245  if (mode & GPIO_MODE_DEF_OUTPUT) {
246  gpio_output_enable(gpio_num);
247  } else {
248  gpio_output_disable(gpio_num);
249  }
250  if (mode & GPIO_MODE_DEF_OD) {
251  GPIO.pin[gpio_num].pad_driver = 1;
252  } else {
253  GPIO.pin[gpio_num].pad_driver = 0;
254  }
255  return ret;
256 }

258 esp_err_t gpio_config(const gpio_config_t *pGPIOConfig)
259 {
260  uint64_t gpio_pin_mask = (pGPIOConfig->pin_bit_mask);
261  uint32_t io_reg = 0;
262  uint32_t io_num = 0;
263  uint8_t input_en = 0;
264  uint8_t output_en = 0;
265  uint8_t od_en = 0;
266  uint8_t pu_en = 0;
267  uint8_t pd_en = 0;
268  if (pGPIOConfig->pin_bit_mask == 0 || pGPIOConfig->pin_bit_mask >= (((uint64_t) 1) << GPIO_PIN_COUNT)) {
269  ESP_LOGE(GPIO_TAG, "GPIO_PIN mask error ");
270  return ESP_ERR_INVALID_ARG;
271  }
272  if ((pGPIOConfig->mode) & (GPIO_MODE_DEF_OUTPUT)) {
273  //GPIO 34/35/36/37/38/39 can only be used as input mode;
274  if ((gpio_pin_mask & ( GPIO_SEL_34 | GPIO_SEL_35 | GPIO_SEL_36 | GPIO_SEL_37 | GPIO_SEL_38 | GPIO_SEL_39))) {
275  ESP_LOGE(GPIO_TAG, "GPIO34-39 can only be used as input mode");
276  return ESP_ERR_INVALID_ARG;
277  }
278  }
279  do {
280  io_reg = GPIO_PIN_MUX_REG[io_num];
281  if (((gpio_pin_mask >> io_num) & BIT(0))) {
282  if (!io_reg) {
283  ESP_LOGE(GPIO_TAG, "IO%d is not a valid GPIO",io_num);
284  return ESP_ERR_INVALID_ARG;
285  }
286  if(RTC_GPIO_IS_VALID_GPIO(io_num)){
287  rtc_gpio_deinit(io_num);
288  }
289  if ((pGPIOConfig->mode) & GPIO_MODE_DEF_INPUT) {
290  input_en = 1;
291  PIN_INPUT_ENABLE(GPIO_PIN_MUX_REG[io_num]);
292  } else {
293  PIN_INPUT_DISABLE(GPIO_PIN_MUX_REG[io_num]);
294  }
295  if ((pGPIOConfig->mode) & GPIO_MODE_DEF_OD) {
296  od_en = 1;
297  GPIO.pin[io_num].pad_driver = 1; /*0x01 Open-drain */
298  } else {
299  GPIO.pin[io_num].pad_driver = 0; /*0x00 Normal gpio output */
300  }
301  if ((pGPIOConfig->mode) & GPIO_MODE_DEF_OUTPUT) {
302  output_en = 1;
303  gpio_output_enable(io_num);
304  } else {
305  gpio_output_disable(io_num);
306  }
307  if (pGPIOConfig->pull_up_en) {
308  pu_en = 1;
309  gpio_pullup_en(io_num);
310  } else {
311  gpio_pullup_dis(io_num);
312  }
313  if (pGPIOConfig->pull_down_en) {
314  pd_en = 1;
315  gpio_pulldown_en(io_num);
316  } else {
317  gpio_pulldown_dis(io_num);
318  }
319  ESP_LOGI(GPIO_TAG, "GPIO[%d]| InputEn: %d| OutputEn: %d| OpenDrain: %d| Pullup: %d| Pulldown: %d| Intr:%d ", io_num, input_en, output_en, od_en, pu_en, pd_en, pGPIOConfig->intr_type);
320  gpio_set_intr_type(io_num, pGPIOConfig->intr_type);
321  if (pGPIOConfig->intr_type) {
322  gpio_intr_enable(io_num);
323  } else {
324  gpio_intr_disable(io_num);
325  }
326  PIN_FUNC_SELECT(io_reg, PIN_FUNC_GPIO); /*function number 2 is GPIO_FUNC for each pin */
327  }
328  io_num++;
329  } while (io_num < GPIO_PIN_COUNT);
330  return ESP_OK;
331 }

333 esp_err_t gpio_reset_pin(gpio_num_t gpio_num)
334 {
335  assert(gpio_num >= 0 && GPIO_IS_VALID_GPIO(gpio_num));
336  gpio_config_t cfg = {
337  .pin_bit_mask = BIT64(gpio_num),
338  .mode = GPIO_MODE_DISABLE,

348 void IRAM_ATTR gpio_intr_service(void* arg)
349 {
350  //GPIO intr process
351  uint32_t gpio_num = 0;
352  //read status to get interrupt status for GPIO0-31
353  const uint32_t gpio_intr_status = (isr_core_id == 0) ? GPIO.pcpu_int : GPIO.acpu_int;
354  //read status1 to get interrupt status for GPIO32-39
355  const uint32_t gpio_intr_status_h = (isr_core_id == 0) ? GPIO.pcpu_int1.intr : GPIO.acpu_int1.intr;
356 
357  if (gpio_isr_func == NULL) {
358  return;
359  }
360  do {
361  if (gpio_num < 32) {
362  if (gpio_intr_status & BIT(gpio_num)) { //gpio0-gpio31
363  if (gpio_isr_func[gpio_num].fn != NULL) {
364  gpio_isr_func[gpio_num].fn(gpio_isr_func[gpio_num].args);
365  }
366  GPIO.status_w1tc = BIT(gpio_num);
367  }
368  } else {
369  if (gpio_intr_status_h & BIT(gpio_num - 32)) {
370  if (gpio_isr_func[gpio_num].fn != NULL) {
371  gpio_isr_func[gpio_num].fn(gpio_isr_func[gpio_num].args);
372  }
373  GPIO.status1_w1tc.intr_st = BIT(gpio_num - 32);
374  }
375  }
376  } while (++gpio_num < GPIO_PIN_COUNT);
377 }

408 esp_err_t gpio_install_isr_service(int intr_alloc_flags)
409 {
410  GPIO_CHECK(gpio_isr_func == NULL, "GPIO isr service already installed", ESP_ERR_INVALID_STATE);
411  esp_err_t ret;
412  portENTER_CRITICAL(&gpio_spinlock);
413  gpio_isr_func = (gpio_isr_func_t*) calloc(GPIO_NUM_MAX, sizeof(gpio_isr_func_t));
414  portEXIT_CRITICAL(&gpio_spinlock);
415  if (gpio_isr_func == NULL) {
416  ret = ESP_ERR_NO_MEM;
417  } else {
418  ret = gpio_isr_register(gpio_intr_service, NULL, intr_alloc_flags, &gpio_isr_handle);
419  }
420  return ret;
421 }

444 esp_err_t gpio_isr_register(void (*fn)(void*), void * arg, int intr_alloc_flags, gpio_isr_handle_t *handle)
445 {
446  GPIO_CHECK(fn, "GPIO ISR null", ESP_ERR_INVALID_ARG);
447  gpio_isr_alloc_t p;
448  p.source = ETS_GPIO_INTR_SOURCE;
449  p.intr_alloc_flags = intr_alloc_flags;
450  p.fn = fn;
451  p.arg = arg;
452  p.handle = handle;
453  portENTER_CRITICAL(&gpio_spinlock);
454  if(isr_core_id == GPIO_ISR_CORE_ID_UNINIT) {
455  isr_core_id = xPortGetCoreID();
456  }
457  portEXIT_CRITICAL(&gpio_spinlock);
458  esp_err_t ret = esp_ipc_call_blocking(isr_core_id, gpio_isr_register_on_core_static, (void *)&p);
459  if(ret != ESP_OK || p.ret != ESP_OK) {
460  return ESP_ERR_NOT_FOUND;
461  }
462  return ESP_OK;
463 }

553 esp_err_t gpio_hold_en(gpio_num_t gpio_num)
554 {
