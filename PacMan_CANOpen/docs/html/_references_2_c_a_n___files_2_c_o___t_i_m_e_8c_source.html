28 /*
29  * Read received message from CAN module.
30  *
31  * Function will be called (by CAN receive interrupt) every time, when CAN
32  * message with correct identifier will be received.
33  */
34 static void CO_TIME_receive(void *object, const CO_CANrxMsg_t *msg){
35  CO_TIME_t *TIME;
36  uint8_t operState;
37 
38  TIME = (CO_TIME_t*)object; /* this is the correct pointer type of the first argument */
39  operState = *TIME->operatingState;
40 
41  if((operState == CO_NMT_OPERATIONAL) || (operState == CO_NMT_PRE_OPERATIONAL)){
42  SET_CANrxNew(TIME->CANrxNew);
43  // Process Time from msg buffer
44  CO_memcpy((uint8_t*)&TIME->Time.ullValue, msg->data, msg->DLC);
45  }
46  else{
47  TIME->receiveError = (uint16_t)msg->DLC;
48  }
49 }

91  TIME->CANdevRxIdx = CANdevRxIdx;

119 /******************************************************************************/
120 uint8_t CO_TIME_process(
121  CO_TIME_t *TIME,
122  uint32_t timeDifference_ms)
123 {
124  uint8_t ret = 0;
125  uint32_t timerNew;
126 
127  if(*TIME->operatingState == CO_NMT_OPERATIONAL || *TIME->operatingState == CO_NMT_PRE_OPERATIONAL){
128  /* update TIME timer, no overflow */
129  timerNew = TIME->timer + timeDifference_ms;
130  if(timerNew > TIME->timer)
131  TIME->timer = timerNew;
132 
133  /* was TIME just received */
134  if(TIME->CANrxNew){
135  TIME->timer = 0;
136  ret = 1;
137  CLEAR_CANrxNew(TIME->CANrxNew);
138  }
139 
140  /* TIME producer */
141  if(TIME->isProducer && TIME->periodTime){
142  if(TIME->timer >= TIME->periodTime){
143  TIME->timer = 0;
144  ret = 1;
145  CO_memcpy(TIME->TXbuff->data, (const uint8_t*)&TIME->Time.ullValue, TIME_MSG_LENGTH);
146  CO_CANsend(TIME->CANdevTx, TIME->TXbuff);
147  }
148  }
149 
150  /* Verify TIME timeout if node is consumer */
151  if(TIME->isConsumer && TIME->periodTime && TIME->timer > TIME->periodTimeoutTime
152  && *TIME->operatingState == CO_NMT_OPERATIONAL)
153  CO_errorReport(TIME->em, CO_EM_TIME_TIMEOUT, CO_EMC_COMMUNICATION, TIME->timer);
154  }
155  else {
156  CLEAR_CANrxNew(TIME->CANrxNew);
157  }
158 
159  /* verify error from receive function */
160  if(TIME->receiveError != 0U){
161  CO_errorReport(TIME->em, CO_EM_TIME_LENGTH, CO_EMC_TIME_DATA_LENGTH, (uint32_t)TIME->receiveError);
162  TIME->receiveError = 0U;
163  }
164 
165  return ret;
166 }
