676  maxFillLen;

691  // is all we have to work with. SPIClass does contain

844  // must be zero when using async!

846  ev.reg = 0;

854  // those are done in the pixel-writing functions.
855  descriptor[d].BTCTRL.bit.VALID = true;
856  // Event strobe on beat xfer:
857  descriptor[d].BTCTRL.bit.EVOSEL = 0x3;
858  descriptor[d].BTCTRL.bit.BLOCKACT =
859  DMA_BLOCK_ACTION_NOACT;
860  descriptor[d].BTCTRL.bit.BEATSIZE = tft8.wide ?
861  DMA_BEAT_SIZE_HWORD : DMA_BEAT_SIZE_BYTE;
862  descriptor[d].BTCTRL.bit.SRCINC = 1;
863  descriptor[d].BTCTRL.bit.DSTINC = 0;
864  descriptor[d].BTCTRL.bit.STEPSEL =
865  DMA_STEPSEL_SRC;
866  descriptor[d].BTCTRL.bit.STEPSIZE =
867  DMA_ADDRESS_INCREMENT_STEP_SIZE_1;
868  descriptor[d].DSTADDR.reg =
869  (uint32_t)tft8.writePort;
870  }
871  #endif // __SAMD51
872  } // end parallel-specific DMA setup
873 
874  lastFillColor = 0x0000;
875  lastFillLen = 0;
876  dma.setCallback(dma_callback);
877  return; // Success!
878  // else clean up any partial allocation...
879  } // end descriptor memalign()

977 void Adafruit_SPITFT::writePixels(uint16_t *colors, uint32_t len,
978  bool block, bool bigEndian) {
979 
980  if(!len) return; // Avoid 0-byte transfers
981 
982 #if defined(ESP32) // ESP32 has a special SPI pixel-writing function...
983  if(connection == TFT_HARD_SPI) {
984  hwspi._spi->writePixels(colors, len * 2);
985  return;
986  }
987 #elif defined(USE_SPI_DMA)
988  if((connection == TFT_HARD_SPI) || (connection == TFT_PARALLEL)) {
989  int maxSpan = maxFillLen / 2; // One scanline max
990  uint8_t pixelBufIdx = 0; // Active pixel buffer number
991  #if defined(__SAMD51__)
992  if(connection == TFT_PARALLEL) {
993  // Switch WR pin to PWM or CCL
994  pinPeripheral(tft8._wr, wrPeripheral);
995  }
996  #endif // end __SAMD51__
997  if(!bigEndian) { // Normal little-endian situation...
998  while(len) {
999  int count = (len < maxSpan) ? len : maxSpan;
1000 
1001  // Because TFT and SAMD endianisms are different, must swap
1002  // bytes from the 'colors' array passed into a DMA working
1003  // buffer. This can take place while the prior DMA transfer
1004  // is in progress, hence the need for two pixelBufs.
1005  for(int i=0; i<count; i++) {
1006  pixelBuf[pixelBufIdx][i] = __builtin_bswap16(*colors++);
1007  }
1008  // The transfers themselves are relatively small, so we don't
1009  // need a long descriptor list. We just alternate between the
1010  // first two, sharing pixelBufIdx for that purpose.
1011  descriptor[pixelBufIdx].SRCADDR.reg =
1012  (uint32_t)pixelBuf[pixelBufIdx] + count * 2;
1013  descriptor[pixelBufIdx].BTCTRL.bit.SRCINC = 1;
1014  descriptor[pixelBufIdx].BTCNT.reg = count * 2;
1015  descriptor[pixelBufIdx].DESCADDR.reg = 0;
1016 
1017  while(dma_busy); // Wait for prior line to finish
1018 
1019  // Move new descriptor into place...
1020  memcpy(dptr, &descriptor[pixelBufIdx], sizeof(DmacDescriptor));
1021  dma_busy = true;
1022  dma.startJob(); // Trigger SPI DMA transfer
1023  if(connection == TFT_PARALLEL) dma.trigger();
1024  pixelBufIdx = 1 - pixelBufIdx; // Swap DMA pixel buffers
1025 
1026  len -= count;
1027  }
1028  } else { // bigEndian == true
1029  // With big-endian pixel data, this can be handled as a single
1030  // DMA transfer using chained descriptors. Even full screen, this
1031  // needs only a relatively short descriptor list, each
1032  // transferring a max of 32,767 (not 32,768) pixels. The list
1033  // was allocated large enough to accommodate a full screen's
1034  // worth of data, so this won't run past the end of the list.
1035  int d, numDescriptors = (len + 32766) / 32767;
1036  for(d=0; d<numDescriptors; d++) {
1037  int count = (len < 32767) ? len : 32767;
1038  descriptor[d].SRCADDR.reg = (uint32_t)colors + count * 2;
1039  descriptor[d].BTCTRL.bit.SRCINC = 1;
1040  descriptor[d].BTCNT.reg = count * 2;
1041  descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d+1];
1042  len -= count;
1043  colors += count;
1044  }
1045  descriptor[d-1].DESCADDR.reg = 0;
1046 
1047  while(dma_busy); // Wait for prior transfer (if any) to finish
1048 
1049  // Move first descriptor into place and start transfer...
1050  memcpy(dptr, &descriptor[0], sizeof(DmacDescriptor));
1051  dma_busy = true;
1052  dma.startJob(); // Trigger SPI DMA transfer
1053  if(connection == TFT_PARALLEL) dma.trigger();
1054  } // end bigEndian
1055 
1056  lastFillColor = 0x0000; // pixelBuf has been sullied
1057  lastFillLen = 0;
1058  if(block) {
1059  while(dma_busy); // Wait for last line to complete
1060  #if defined(__SAMD51__) || defined(_SAMD21_)
1061  if(connection == TFT_HARD_SPI) {
1062  // See SAMD51/21 note in writeColor()
1063  hwspi._spi->setDataMode(hwspi._mode);
1064  } else {
1065  pinPeripheral(tft8._wr, PIO_OUTPUT); // Switch WR back to GPIO
1066  }
1067  #endif // end __SAMD51__ || _SAMD21_
1068  }
1069  return;
1070  }
1071 #endif // end USE_SPI_DMA
1072 
1073  // All other cases (bitbang SPI or non-DMA hard SPI or parallel),
1074  // use a loop with the normal 16-bit data write function:
1075  while(len--) {
1076  SPI_WRITE16(*colors++);
1077  }
1078 }

1106 void Adafruit_SPITFT::writeColor(uint16_t color, uint32_t len) {
1107 
1108  if(!len) return; // Avoid 0-byte transfers
1109 
1110  uint8_t hi = color >> 8, lo = color;
1111 
1112 #if defined(ESP32) // ESP32 has a special SPI pixel-writing function...
1113  if(connection == TFT_HARD_SPI) {
1114  #define SPI_MAX_PIXELS_AT_ONCE 32
1115  #define TMPBUF_LONGWORDS (SPI_MAX_PIXELS_AT_ONCE + 1) / 2
1116  #define TMPBUF_PIXELS (TMPBUF_LONGWORDS * 2)
1117  static uint32_t temp[TMPBUF_LONGWORDS];
1118  uint32_t c32 = color * 0x00010001;
1119  uint16_t bufLen = (len < TMPBUF_PIXELS) ? len : TMPBUF_PIXELS,
1120  xferLen, fillLen;
1121  // Fill temp buffer 32 bits at a time
1122  fillLen = (bufLen + 1) / 2; // Round up to next 32-bit boundary
1123  for(uint32_t t=0; t<fillLen; t++) {
1124  temp[t] = c32;
1125  }
1126  // Issue pixels in blocks from temp buffer
1127  while(len) { // While pixels remain
1128  xferLen = (bufLen < len) ? bufLen : len; // How many this pass?
1129  writePixels((uint16_t *)temp, xferLen);
1130  len -= xferLen;
1131  }
1132  return;
1133  }
1134 #else // !ESP32
1135  #if defined(USE_SPI_DMA)
1136  if(((connection == TFT_HARD_SPI) || (connection == TFT_PARALLEL)) &&
1137  (len >= 16)) { // Don't bother with DMA on short pixel runs
1138  int i, d, numDescriptors;
1139  if(hi == lo) { // If high & low bytes are same...
1140  onePixelBuf = color;
1141  // Can do this with a relatively short descriptor list,
1142  // each transferring a max of 32,767 (not 32,768) pixels.
1143  // This won't run off the end of the allocated descriptor list,
1144  // since we're using much larger chunks per descriptor here.
1145  numDescriptors = (len + 32766) / 32767;
1146  for(d=0; d<numDescriptors; d++) {
1147  int count = (len < 32767) ? len : 32767;
1148  descriptor[d].SRCADDR.reg = (uint32_t)&onePixelBuf;
1149  descriptor[d].BTCTRL.bit.SRCINC = 0;
1150  descriptor[d].BTCNT.reg = count * 2;
1151  descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d+1];
1152  len -= count;
1153  }
1154  descriptor[d-1].DESCADDR.reg = 0;
1155  } else {
1156  // If high and low bytes are distinct, it's necessary to fill
1157  // a buffer with pixel data (swapping high and low bytes because
1158  // TFT and SAMD are different endianisms) and create a longer
1159  // descriptor list pointing repeatedly to this data. We can do
1160  // this slightly faster working 2 pixels (32 bits) at a time.
1161  uint32_t *pixelPtr = (uint32_t *)pixelBuf[0],
1162  twoPixels = __builtin_bswap16(color) * 0x00010001;
1163  // We can avoid some or all of the buffer-filling if the color
1164  // is the same as last time...
1165  if(color == lastFillColor) {
1166  // If length is longer than prior instance, fill only the
1167  // additional pixels in the buffer and update lastFillLen.
1168  if(len > lastFillLen) {
1169  int fillStart = lastFillLen / 2,
1170  fillEnd = (((len < maxFillLen) ?
1171  len : maxFillLen) + 1) / 2;
1172  for(i=fillStart; i<fillEnd; i++) pixelPtr[i] = twoPixels;
1173  lastFillLen = fillEnd * 2;
1174  } // else do nothing, don't set pixels or change lastFillLen
1175  } else {
1176  int fillEnd = (((len < maxFillLen) ?
1177  len : maxFillLen) + 1) / 2;
1178  for(i=0; i<fillEnd; i++) pixelPtr[i] = twoPixels;
1179  lastFillLen = fillEnd * 2;
1180  lastFillColor = color;
1181  }
1182 
1183  numDescriptors = (len + maxFillLen - 1) / maxFillLen;
1184  for(d=0; d<numDescriptors; d++) {
1185  int pixels = (len < maxFillLen) ? len : maxFillLen,
1186  bytes = pixels * 2;
1187  descriptor[d].SRCADDR.reg = (uint32_t)pixelPtr + bytes;
1188  descriptor[d].BTCTRL.bit.SRCINC = 1;
1189  descriptor[d].BTCNT.reg = bytes;
1190  descriptor[d].DESCADDR.reg = (uint32_t)&descriptor[d+1];
1191  len -= pixels;
1192  }
1193  descriptor[d-1].DESCADDR.reg = 0;
1194  }
1195  memcpy(dptr, &descriptor[0], sizeof(DmacDescriptor));
1196  #if defined(__SAMD51__)
1197  if(connection == TFT_PARALLEL) {
1198  // Switch WR pin to PWM or CCL
1199  pinPeripheral(tft8._wr, wrPeripheral);
1200  }
1201  #endif // end __SAMD51__
1202 
1203  dma_busy = true;
1204  dma.startJob();
1205  if(connection == TFT_PARALLEL) dma.trigger();
1206  while(dma_busy); // Wait for completion
1207  // Unfortunately blocking is necessary. An earlier version returned
1208  // immediately and checked dma_busy on startWrite() instead, but it
1209  // turns out to be MUCH slower on many graphics operations (as when
1210  // drawing lines, pixel-by-pixel), perhaps because it's a volatile
1211  // type and doesn't cache. Working on this.
1212  #if defined(__SAMD51__) || defined(_SAMD21_)
1213  if(connection == TFT_HARD_SPI) {
1214  // SAMD51: SPI DMA seems to leave the SPI peripheral in a freaky
1215  // state on completion. Workaround is to explicitly set it back...
1216  // (5/17/2019: apparently SAMD21 too, in certain cases, observed
1217  // with ST7789 display.)
1218  hwspi._spi->setDataMode(hwspi._mode);
1219  } else {
1220  pinPeripheral(tft8._wr, PIO_OUTPUT); // Switch WR back to GPIO
1221  }
1222  #endif // end __SAMD51__
1223  return;
1224  }
1225  #endif // end USE_SPI_DMA
