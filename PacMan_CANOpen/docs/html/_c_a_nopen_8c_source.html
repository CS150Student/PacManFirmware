184  CO_ReturnError_t CO_sendNMTcommand(CO_t *co, uint8_t command, uint8_t nodeID){
185  if(NMTM_txBuff == 0){
186  /* error, CO_CANtxBufferInit() was not called for this buffer. */
187  return CO_ERROR_TX_UNCONFIGURED; /* -11 */
188  }
189  NMTM_txBuff->data[0] = command;
190  NMTM_txBuff->data[1] = nodeID;
191 
192  CO_ReturnError_t error = CO_ERROR_NO;
193 
194  /* Apply NMT command also to this node, if set so. */
195  if(nodeID == 0 || nodeID == co->NMT->nodeId){
196  switch(command){
197  case CO_NMT_ENTER_OPERATIONAL:
198  if((*co->NMT->emPr->errorRegister) == 0) {
199  co->NMT->operatingState = CO_NMT_OPERATIONAL;
200  }
201  break;
202  case CO_NMT_ENTER_STOPPED:
203  co->NMT->operatingState = CO_NMT_STOPPED;
204  break;
205  case CO_NMT_ENTER_PRE_OPERATIONAL:
206  co->NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
207  break;
208  case CO_NMT_RESET_NODE:
209  co->NMT->resetCommand = CO_RESET_APP;
210  break;
211  case CO_NMT_RESET_COMMUNICATION:
212  co->NMT->resetCommand = CO_RESET_COMM;
213  break;
214  default:
215  error = CO_ERROR_ILLEGAL_ARGUMENT;
216  break;
217 
218  }
219  }
220 
221  if(error == CO_ERROR_NO)
222  return CO_CANsend(co->CANmodule[0], NMTM_txBuff); /* 0 = success */
223  else
224  {
225  return error;
226  }
227

493 /******************************************************************************/
494 CO_ReturnError_t CO_CANopenInit(
495  uint8_t nodeId)
496 {
497  int16_t i;
498  CO_ReturnError_t err;
499 
500  /* Verify CANopen Node-ID */
501  if(nodeId<1 || nodeId>127) {
502  return CO_ERROR_PARAMETERS;
503  }
504 
505  for (i=0; i<CO_NO_SDO_SERVER; i++)
506  {
507  uint32_t COB_IDClientToServer;
508  uint32_t COB_IDServerToClient;
509  if(i==0){
510  /*Default SDO server must be located at first index*/
511  COB_IDClientToServer = CO_CAN_ID_RSDO + nodeId;
512  COB_IDServerToClient = CO_CAN_ID_TSDO + nodeId;
513  }else{
514  COB_IDClientToServer = OD_SDOServerParameter[i].COB_IDClientToServer;
515  COB_IDServerToClient = OD_SDOServerParameter[i].COB_IDServerToClient;
516  }
517 
518  err = CO_SDO_init(
519  CO->SDO[i],
520  COB_IDClientToServer,
521  COB_IDServerToClient,
522  OD_H1200_SDO_SERVER_PARAM+i,
523  i==0 ? 0 : CO->SDO[0],
524  &CO_OD[0],
525  CO_OD_NoOfElements,
526  CO_SDO_ODExtensions,
527  nodeId,
528  CO->CANmodule[0],
529  CO_RXCAN_SDO_SRV+i,
