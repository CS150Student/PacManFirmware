208  CO_CANmodule_t *CANmodule,

220 
221  /* CAN identifier, DLC and rtr, bit aligned with CAN module transmit buffer.

225  buffer->ident = ((uint32_t)ident & 0x07FFU)

227  | ((uint32_t)(rtr ? 0x8000U : 0U));

237 CO_ReturnError_t CO_CANsend(CO_CANmodule_t *CANmodule, CO_CANtx_t *buffer) {
238  CO_ReturnError_t err = CO_ERROR_NO;
239 
240  /* Verify overflow */
241  if (buffer->bufferFull) {
242  if (!CANmodule->firstCANtxMessage) {
243  /* don't set error, if bootup message is still on buffers */
244  CO_errorReport((CO_EM_t*)CANmodule->em, CO_EM_CAN_TX_OVERFLOW, CO_EMC_CAN_OVERRUN, buffer->ident);
245  }
246  err = CO_ERROR_TX_OVERFLOW;
247  }
248  // Starting to create message to use the ESP IDF Send Function
249  can_message_t message;
250  message.identifier = buffer->esp_ident;
251  message.extd = 0;
252  message.data_length_code = CAN_FRAME_MAX_DLC;
253 
254  for (int i = 0; i < CAN_FRAME_MAX_DLC; i++) {
255  message.data[i] = buffer->data[i];
256  //printf("Message Contents: %u \n", message.data[i]);
257  }
258 
259  CO_LOCK_CAN_SEND();
260  /* if CAN TX buffer is free, copy message to it */
261  if (1 && CANmodule->CANtxCount == 0) {
262  CANmodule->bufferInhibitFlag = buffer->syncFlag;
263 
264  //Queue message for transmission
265  if (can_transmit(&message, pdMS_TO_TICKS(1000)) == ESP_OK) {
266  printf("Message queued for transmission\n");
267  } else {
268  printf("\nFailed to queue message for transmission\n");
269  }
270  }
271  /* if no buffer is free, message will be sent by interrupt */
272  else {
273  buffer->bufferFull = true;
274  CANmodule->CANtxCount++;
275  }
276  CO_UNLOCK_CAN_SEND();
277 
278  return err;
279 }

318 /******************************************************************************/
319 void CO_CANverifyErrors(CO_CANmodule_t *CANmodule) {
320  uint16_t rxErrors, txErrors, overflow;
321  CO_EM_t* em = (CO_EM_t*)CANmodule->em;
322  uint32_t err;
323 
324  /* get error counters from module. Id possible, function may use different way to
325  determine errors. */
326  rxErrors = CANmodule->txSize;
327  txErrors = CANmodule->txSize;
328  overflow = CANmodule->txSize;
329 
330  err = ((uint32_t)txErrors << 16) | ((uint32_t)rxErrors << 8) | overflow;
331 
332  if (CANmodule->errOld != err) {
333  CANmodule->errOld = err;
334 
335  if (txErrors >= 256U) { /* bus off */
336  CO_errorReport(em, CO_EM_CAN_TX_BUS_OFF, CO_EMC_BUS_OFF_RECOVERED, err);
337  }
338  else { /* not bus off */
339  CO_errorReset(em, CO_EM_CAN_TX_BUS_OFF, err);
340 
341  if ((rxErrors >= 96U) || (txErrors >= 96U)) { /* bus warning */
342  CO_errorReport(em, CO_EM_CAN_BUS_WARNING, CO_EMC_NO_ERROR, err);
343  }
344 
345  if (rxErrors >= 128U) { /* RX bus passive */
346  CO_errorReport(em, CO_EM_CAN_RX_BUS_PASSIVE, CO_EMC_CAN_PASSIVE, err);
347  }
348  else {
349  CO_errorReset(em, CO_EM_CAN_RX_BUS_PASSIVE, err);
350  }
351 
352  if (txErrors >= 128U) { /* TX bus passive */
353  if (!CANmodule->firstCANtxMessage) {
354  CO_errorReport(em, CO_EM_CAN_TX_BUS_PASSIVE, CO_EMC_CAN_PASSIVE, err);
355  }
356  }
357  else {
358  bool_t isError = CO_isError(em, CO_EM_CAN_TX_BUS_PASSIVE);
359  if (isError) {
360  CO_errorReset(em, CO_EM_CAN_TX_BUS_PASSIVE, err);
361  CO_errorReset(em, CO_EM_CAN_TX_OVERFLOW, err);
362  }
363  }
364 
365  if ((rxErrors < 96U) && (txErrors < 96U)) { /* no error */
366  CO_errorReset(em, CO_EM_CAN_BUS_WARNING, err);
367  }
368  }
369 
370  if (overflow != 0U) { /* CAN RX bus overflow */
371  CO_errorReport(em, CO_EM_CAN_RXB_OVERFLOW, CO_EMC_CAN_OVERRUN, err);
372  }
373  }
