32 /*
33  * Read received message from CAN module.
34  *
35  * Function will be called (by CAN receive interrupt) every time, when CAN
36  * message with correct identifier will be received. For more information and
37  * description of parameters see file CO_driver.h.
38  */
39 static void CO_NMT_receive(void *object, const CO_CANrxMsg_t *msg){
40  CO_NMT_t *NMT;
41  uint8_t nodeId;
42 
43  NMT = (CO_NMT_t*)object; /* this is the correct pointer type of the first argument */
44 
45  nodeId = msg->data[1];
46 
47  if((msg->DLC == 2) && ((nodeId == 0) || (nodeId == NMT->nodeId))){
48  uint8_t command = msg->data[0];
49  uint8_t currentOperatingState = NMT->operatingState;
50 
51  switch(command){
52  case CO_NMT_ENTER_OPERATIONAL:
53  if((*NMT->emPr->errorRegister) == 0U){
54  NMT->operatingState = CO_NMT_OPERATIONAL;
55  }
56  break;
57  case CO_NMT_ENTER_STOPPED:
58  NMT->operatingState = CO_NMT_STOPPED;
59  break;
60  case CO_NMT_ENTER_PRE_OPERATIONAL:
61  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
62  break;
63  case CO_NMT_RESET_NODE:
64  NMT->resetCommand = CO_RESET_APP;
65  break;
66  case CO_NMT_RESET_COMMUNICATION:
67  NMT->resetCommand = CO_RESET_COMM;
68  break;
69  default:
70  break;
71  }
72 
73  if(NMT->pFunctNMT!=NULL && currentOperatingState!=NMT->operatingState){
74  NMT->pFunctNMT(NMT->operatingState);
75  }
76  }
77 }

80 /******************************************************************************/
81 CO_ReturnError_t CO_NMT_init(
82  CO_NMT_t *NMT,
83  CO_EMpr_t *emPr,
84  uint8_t nodeId,
85  uint16_t firstHBTime,
86  CO_CANmodule_t *NMT_CANdev,
87  uint16_t NMT_rxIdx,
88  uint16_t CANidRxNMT,
89  CO_CANmodule_t *HB_CANdev,
90  uint16_t HB_txIdx,
91  uint16_t CANidTxHB)
92 {
93  /* verify arguments */
94  if(NMT==NULL || emPr==NULL || NMT_CANdev==NULL || HB_CANdev==NULL){
95  return CO_ERROR_ILLEGAL_ARGUMENT;
96  }
97 
98  /* blinking bytes */
99 #ifdef CO_USE_LEDS
100  NMT->LEDflickering = 0;
101  NMT->LEDblinking = 0;
102  NMT->LEDsingleFlash = 0;
103  NMT->LEDdoubleFlash = 0;
104  NMT->LEDtripleFlash = 0;
105  NMT->LEDquadrupleFlash = 0;
106 #endif /* CO_USE_LEDS */
107 
108  /* Configure object variables */
109  NMT->operatingState = CO_NMT_INITIALIZING;
110 #ifdef CO_USE_LEDS
111  NMT->LEDgreenRun = -1;
112  NMT->LEDredError = 1;
113 #endif /* CO_USE_LEDS */
114  NMT->nodeId = nodeId;
115  NMT->firstHBTime = firstHBTime;
116  NMT->resetCommand = 0;
117  NMT->HBproducerTimer = 0xFFFF;
118  NMT->emPr = emPr;
119  NMT->pFunctNMT = NULL;
120 
121  /* configure NMT CAN reception */
122  printf("\nCONMT Print: %u ", NMT_rxIdx);
123  CO_CANrxBufferInit(
124  NMT_CANdev, /* CAN device */
125  NMT_rxIdx, /* rx buffer index */
126  CANidRxNMT, /* CAN identifier */
127  0x7FF, /* mask */
128  0, /* rtr */
129  (void*)NMT, /* object passed to receive function */
130  CO_NMT_receive); /* this function will process received message */
131 
132  /* configure HB CAN transmission */
133  NMT->HB_CANdev = HB_CANdev;
134  NMT->HB_TXbuff = CO_CANtxBufferInit(
135  HB_CANdev, /* CAN device */
136  HB_txIdx, /* index of specific buffer inside CAN module */
137  CANidTxHB, /* CAN identifier */
138  0, /* rtr */
139  1, /* number of data bytes */
140  0); /* synchronous message flag bit */
141 
142  return CO_ERROR_NO;
143 }

200 /******************************************************************************/
201 CO_NMT_reset_cmd_t CO_NMT_process(
202  CO_NMT_t *NMT,
203  uint16_t timeDifference_ms,
204  uint16_t HBtime,
205  uint32_t NMTstartup,
206  uint8_t errorRegister,
207  const uint8_t errorBehavior[],
208  uint16_t *timerNext_ms)
209 {
210  uint8_t CANpassive;
211 
212  uint8_t currentOperatingState = NMT->operatingState;
213 
214  NMT->HBproducerTimer += timeDifference_ms;
215 
216  /* Heartbeat producer message & Bootup message */
217  if((HBtime != 0 && NMT->HBproducerTimer >= HBtime) || NMT->operatingState == CO_NMT_INITIALIZING){
218 
219  /* Start from the beginning. If OS is slow, time sliding may occur. However, heartbeat is
220  * not for synchronization, it is for health report. */
221  NMT->HBproducerTimer = 0;
222 
223  NMT->HB_TXbuff->data[0] = NMT->operatingState;
224  CO_CANsend(NMT->HB_CANdev, NMT->HB_TXbuff);
225 
226  if(NMT->operatingState == CO_NMT_INITIALIZING){
227  if(HBtime > NMT->firstHBTime) NMT->HBproducerTimer = HBtime - NMT->firstHBTime;
228  else NMT->HBproducerTimer = 0;
229 
230  /* NMT slave self starting */
231  if (NMTstartup == 0x00000008U) NMT->operatingState = CO_NMT_OPERATIONAL;
232  else NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
233  }
234  }
235 
236 
237  /* Calculate, when next Heartbeat needs to be send and lower timerNext_ms if necessary. */
238  if(HBtime != 0 && timerNext_ms != NULL){
239  if(NMT->HBproducerTimer < HBtime){
240  uint16_t diff = HBtime - NMT->HBproducerTimer;
241  if(*timerNext_ms > diff){
242  *timerNext_ms = diff;
243  }
244  }else{
245  *timerNext_ms = 0;
246  }
247  }
248 
249 
250  /* CAN passive flag */
251  CANpassive = 0;
252  if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_PASSIVE) || CO_isError(NMT->emPr->em, CO_EM_CAN_RX_BUS_PASSIVE))
253  CANpassive = 1;
254 
255 
256 #ifdef CO_USE_LEDS
257  /* CANopen green RUN LED (DR 303-3) */
258  switch(NMT->operatingState){
259  case CO_NMT_STOPPED: NMT->LEDgreenRun = NMT->LEDsingleFlash; break;
260  case CO_NMT_PRE_OPERATIONAL: NMT->LEDgreenRun = NMT->LEDblinking; break;
261  case CO_NMT_OPERATIONAL: NMT->LEDgreenRun = 1; break;
262  default: break;
263  }
264 
265 
266  /* CANopen red ERROR LED (DR 303-3) */
267  if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_OFF))
268  NMT->LEDredError = 1;
269 
270  else if(CO_isError(NMT->emPr->em, CO_EM_SYNC_TIME_OUT))
271  NMT->LEDredError = NMT->LEDtripleFlash;
272 
273  else if(CO_isError(NMT->emPr->em, CO_EM_HEARTBEAT_CONSUMER) || CO_isError(NMT->emPr->em, CO_EM_HB_CONSUMER_REMOTE_RESET))
274  NMT->LEDredError = NMT->LEDdoubleFlash;
275 
276  else if(CANpassive || CO_isError(NMT->emPr->em, CO_EM_CAN_BUS_WARNING))
277  NMT->LEDredError = NMT->LEDsingleFlash;
278 
279  else if(errorRegister)
280  NMT->LEDredError = (NMT->LEDblinking>=0)?-1:1;
281 
282  else
283  NMT->LEDredError = -1;
284 #endif /* CO_USE_LEDS */
285 
286 
287  /* in case of error enter pre-operational state */
288  if(errorBehavior && (NMT->operatingState == CO_NMT_OPERATIONAL)){
289  if(CANpassive && (errorBehavior[2] == 0 || errorBehavior[2] == 2)) errorRegister |= 0x10;
290 
291  if(errorRegister){
292  /* Communication error */
293  if(errorRegister & CO_ERR_REG_COMM_ERR){
294  if(errorBehavior[1] == 0){
295  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
296  }
297  else if(errorBehavior[1] == 2){
298  NMT->operatingState = CO_NMT_STOPPED;
299  }
300  else if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_OFF)
301  || CO_isError(NMT->emPr->em, CO_EM_HEARTBEAT_CONSUMER)
302  || CO_isError(NMT->emPr->em, CO_EM_HB_CONSUMER_REMOTE_RESET))
303  {
304  if(errorBehavior[0] == 0){
305  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
306  }
307  else if(errorBehavior[0] == 2){
308  NMT->operatingState = CO_NMT_STOPPED;
309  }
310  }
311  }
312 
313  /* Generic error */
314  if(errorRegister & CO_ERR_REG_GENERIC_ERR){
315  if (errorBehavior[3] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
316  else if (errorBehavior[3] == 2) NMT->operatingState = CO_NMT_STOPPED;
317  }
318 
319  /* Device profile error */
320  if(errorRegister & CO_ERR_REG_DEV_PROFILE){
321  if (errorBehavior[4] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
322  else if (errorBehavior[4] == 2) NMT->operatingState = CO_NMT_STOPPED;
323  }
324 
325  /* Manufacturer specific error */
326  if(errorRegister & CO_ERR_REG_MANUFACTURER){
327  if (errorBehavior[5] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
328  else if (errorBehavior[5] == 2) NMT->operatingState = CO_NMT_STOPPED;
329  }
330 
331  /* if operational state is lost, send HB immediately. */
332  if(NMT->operatingState != CO_NMT_OPERATIONAL)
333  NMT->HBproducerTimer = HBtime;
334  }
335  }
336 
337  if(NMT->pFunctNMT!=NULL && currentOperatingState!=NMT->operatingState){
338  NMT->pFunctNMT(NMT->operatingState);
339  }
340 
341  return NMT->resetCommand;
342 }
